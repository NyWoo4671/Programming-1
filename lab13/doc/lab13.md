	for (char *word = strtok(text_copy, space); word != NULL; word = strtok(NULL, space)) {
        int length = strlen(word);

        if (length > max_length) {
            max_length = length;
            max_word = word;
        }

        if (length < min_length) {
            min_length = length;
            min_word = word;
        }
    }
